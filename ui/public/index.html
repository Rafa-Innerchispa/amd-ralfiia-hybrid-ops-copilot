Create script: cat << 'EOF' > gemma_api.py import uvicorn from fastapi import FastAPI from pydantic import BaseModel from transformers import pipeline import torch torch.set_num_threads(4) app = FastAPI() pipe = pipeline("text-generation", model="unsloth/gemma-2-2b-it", torch_dtype=torch.float32, device_map="cpu") class ChatMessage(BaseModel): role: str content: str class ChatRequest(BaseModel): model: str messages: list[ChatMessage] @app.get("/v1/models") async def list_models(): return {"data": [{"id": "unsloth/gemma-2-2b-it"}]} @app.post("/v1/chat/completions") async def chat_completions(req: ChatRequest): prompt = "" for msg in req.messages: prompt += f"{msg.role}\n{msg.content}\n" prompt += "model\n" outputs = pipe(prompt, max_new_tokens=48, do_sample=True, temperature=0.7) return {"choices": [{"message": {"role": "assistant", "content": outputs[0]["generated_text"][len(prompt):].strip()}}]} if __name__ == "__main__": uvicorn.run(app, host="0.0.0.0", port=8000) EOF
  4. Run: python3 gemma_api.py
